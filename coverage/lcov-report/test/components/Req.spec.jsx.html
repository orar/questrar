import React from 'react';
import { shallow } from 'enzyme';
import Request from '../../src/components/Request';
import wrapRequest from './RequestWrap';
import { randomId } from '../../src/module/helper';
import { initialRequest } from '../../src/module/common';
 
const Requestor = () => <div>Test Requestor</div>
 
describe('Wrap a request test', () => {
  let id;
  let wrapper;
  let requestState;
 
  const createWrapper = () => {
    id = randomId();
    requestState = { ...initialRequest, id, success: true }
 
    wrapper = shallow(
      <Request inject>
        <Requestor />
      </Request>,
    )
  };
 
  beforeEach(() => {
    createWrapper();
  });
 
  it('Should rip bare the Request Component', () => {
    const wrap = wrapRequest(wrapper)(requestState);
    expects(wrap.prop('inject')).to.be.true();
    expects(wrap.prop('id')).to.equal(id);
  });
 
  it('Should have the underlying component as immediate child', () => {
    const wrap = wrapRequest(wrapper)(requestState);
    expects(wrap.children().first().is(Requestor)).to.be.true();
  });
});
